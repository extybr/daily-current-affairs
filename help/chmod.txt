Инструкция по chmod

Основной синтаксис:
chmod [опции] режим файл — изменить права доступа к файлу/директории

Форматы задания прав:
1. Символьный (буквенный) формат

Кто:
u — владелец (user)
g — группа (group)
o — остальные (others)
a — все (all = ugo)

Действие:
+ — добавить право
- — убрать право
= — установить точно

Права:
r — чтение (read)
w — запись (write)
x — выполнение (execute)
X — выполнение только для директорий или если уже есть x у кого-то
s — setuid/setgid
t — sticky bit
2. Числовой (восьмеричный) формат

Цифры:
4 — чтение (r)
2 — запись (w)
1 — выполнение (x)

Комбинации:
7 = 4+2+1 — чтение, запись, выполнение
6 = 4+2 — чтение, запись
5 = 4+1 — чтение, выполнение
4 = 4 — только чтение
3 = 2+1 — запись, выполнение
2 = 2 — только запись
1 = 1 — только выполнение
0 = 0 — нет прав

Три цифры:
755 — владелец:7, группа:5, остальные:5
644 — владелец:6, группа:4, остальные:4
Основные опции
    chmod -R           # рекурсивно (для директорий)
    chmod -v           # подробный вывод (verbose)
    chmod -c           # вывод только изменений
    chmod -f           # подавить ошибки (force)
    chmod --reference=ФАЙЛ_ЭТАЛОН ФАЙЛ  # скопировать права с другого файла

Примеры символьного формата
    # Добавление прав
    chmod u+x file              # добавить выполнение для владельца
    chmod g+w file              # добавить запись для группы
    chmod o+r file              # добавить чтение для остальных
    chmod a+x file              # добавить выполнение для всех
    chmod u+rwx file            # добавить все права для владельца

    # Удаление прав
    chmod u-x file              # убрать выполнение у владельца
    chmod g-w file              # убрать запись у группы
    chmod o-r file              # убрать чтение у остальных
    chmod a-x file              # убрать выполнение у всех

    # Точная установка
    chmod u=rwx file            # установить rwx для владельца
    chmod g=rx file             # установить rx для группы
    chmod o=r file              # установить только чтение для остальных
    chmod u=rwx,g=rx,o=r file   # установить всё сразу

    # Комбинирование
    chmod u+rw,g+rw,o+r file    # добавить разные права разным группам
    chmod a+rwx file            # дать все права всем
    chmod a-x file              # убрать выполнение у всех

Примеры числового формата
    # Базовые права
    chmod 755 file              # rwxr-xr-x (владелец: все, группа/остальные: rx)
    chmod 644 file              # rw-r--r-- (владелец: rw, группа/остальные: r)
    chmod 600 file              # rw------- (только владелец读写)
    chmod 700 file              # rwx------ (только владелец все права)

    # Для директорий
    chmod 755 dir               # владелец: всё, остальные: чтение и вход
    chmod 750 dir               # владелец: всё, группа: rx, остальные: нет
    chmod 700 dir               # только владелец

    # Скрипты
    chmod 755 script.sh         # исполняемый для всех
    chmod 700 script.sh         # исполняемый только для владельца

    # Чтение/запись
    chmod 444 file              # только чтение для всех
    chmod 666 file              # чтение и запись для всех
    chmod 777 file              # все права для всех (опасно!)

    # Безопасные настройки
    chmod 640 file              # rw-r----- (владелец: rw, группа: r, остальные: нет)
    chmod 750 file              # rwxr-x--- (владелец: всё, группа: rx, остальные: нет)

Специальные биты
    # Setuid (4xxx) - выполнение от имени владельца файла
    chmod u+s file              # установить setuid
    chmod 4755 file             # rwsr-xr-x
    chmod u-s file              # убрать setuid

    # Setgid (2xxx) - выполнение от имени группы
    chmod g+s file              # установить setgid
    chmod 2755 file             # rwxr-sr-x
    chmod g-s file              # убрать setgid

    # Sticky bit (1xxx) - только владелец может удалять файлы в директории
    chmod +t dir                # установить sticky bit
    chmod 1777 dir              # rwxrwxrwt (как /tmp)
    chmod -t dir                # убрать sticky bit

Рекурсивные операции
    # Изменение прав для всех файлов в директории
    chmod -R 755 dir/           # рекурсивно установить 755

    # Только для директорий (не для файлов)
    find dir/ -type d -exec chmod 755 {} \;

    # Только для файлов
    find dir/ -type f -exec chmod 644 {} \;

    # Раздельная установка
    chmod -R u+rwX,go+rX,go-w dir/   # X - только для директорий и уже исполняемых

Практические примеры
1. Скрипты и программы
    # Сделать скрипт исполняемым
    chmod +x script.sh
    chmod 755 script.sh

    # Установка прав для системных скриптов
    chmod 700 /usr/local/bin/private-script
    chmod 755 /usr/local/bin/public-script

2. Веб-сервер
    # Безопасные права для веб-директории
    chmod 755 /var/www/html
    chmod 644 /var/www/html/*.html
    chmod 640 /var/www/html/config.php
    chmod 750 /var/www/html/cgi-bin/

3. SSH ключи
    # Домашняя директория
    chmod 700 ~/.ssh
    chmod 600 ~/.ssh/id_rsa        # приватный ключ
    chmod 644 ~/.ssh/id_rsa.pub    # публичный ключ
    chmod 600 ~/.ssh/authorized_keys
    chmod 644 ~/.ssh/known_hosts
    chmod 644 ~/.ssh/config

4. Общие ресурсы
    # Директория для совместной работы
    chmod 2775 /shared             # setgid + rwxrwxr-x
    chmod 660 /shared/*            # rw-rw---- для файлов

    # Временная директория
    chmod 1777 /tmp                # sticky bit для безопасности

5. Резервное копирование
    # Копирование прав с одного файла на другой
    chmod --reference=template.txt file.txt

    # Сохранение и восстановление прав
    getfacl file.txt > file.acl
    setfacl --restore=file.acl

Массовые операции
    # Все .sh файлы сделать исполняемыми
    chmod +x *.sh

    # Все .txt файлы сделать только для чтения
    chmod 444 *.txt

    # Рекурсивно убрать права на запись у всех
    chmod -R a-w dir/

    # Дать права на выполнение всем .pl файлам
    find . -name "*.pl" -exec chmod +x {} \;

Комбинации с другими командами
    # Создать файл с правами из umask
    umask 022; touch file.txt    # создаст 644
    umask 077; touch file.txt    # создаст 600

    # Скопировать файл с сохранением прав
    cp -p source.txt dest.txt

    # Изменить права на основе текущих
    chmod u+rwx,g+rx,o-rwx file

Проверка прав
    # Просмотр текущих прав
    ls -l file
    stat -c "%a %n" file         # восьмеричные права
    stat -c "%A %n" file         # символьные права

    # Проверка прав в скриптах
    if [[ -r file ]]; then
        echo "Файл доступен для чтения"
    fi

    if [[ -w file ]]; then
        echo "Файл доступен для записи"
    fi

    if [[ -x file ]]; then
        echo "Файл исполняемый"
    fi

Типовые комбинации прав

Для файлов:
    644  # rw-r--r--  (обычные файлы, документы)
    600  # rw-------  (конфиденциальные файлы)
    755  # rwxr-xr-x  (исполняемые файлы, скрипты)
    700  # rwx------  (приватные скрипты)
    444  # r--r--r--  (только чтение)
    640  # rw-r-----  (файлы для группы)

Для директорий:
    755  # drwxr-xr-x (обычные директории)
    750  # drwxr-x--- (директории для группы)
    700  # drwx------ (приватные директории)
    555  # dr-xr-xr-x (только чтение и вход)
    1777 # drwxrwxrwt (публичные директории, как /tmp)
    2755 # drwxr-sr-x (setgid для наследования группы)

Безопасность
    # Никогда не делайте
    chmod 777 /etc/passwd
    chmod 777 ~/.ssh/id_rsa
    chmod 777 /tmp

    # Проверка опасных прав
    find / -perm -0002 -type f    # файлы, доступные для записи всем
    find / -perm -4000 -type f    # setuid файлы

    # Исправление опасных прав
    chmod 644 /etc/passwd
    chmod 600 ~/.ssh/id_rsa
    chmod 755 /usr/bin/sudo

Маски umask
    # Просмотр текущей umask
    umask                    # 0022

    # Установка umask
    umask 022                # права по умолчанию: файлы 644, директории 755
    umask 077                # права по умолчанию: файлы 600, директории 700

    # Постоянная установка в ~/.bashrc
    echo "umask 027" >> ~/.bashrc

Специальные случаи
    # Права для символических ссылок
    # chmod на symlink меняет права цели, а не ссылки

    # Файлы с кавычками и пробелами
    chmod 755 "file with spaces.txt"
    chmod 755 file\ with\ spaces.txt

    # Массовое изменение с шаблоном
    chmod 755 dir/*/subdir/file.txt

Отладка
    # Подробный вывод
    chmod -v 755 file
    # mode of 'file' changed from 0644 (rw-r--r--) to 0755 (rwxr-xr-x)

    # Только изменения
    chmod -c 755 file

    # Игнорировать ошибки
    chmod -f 755 nonexistent_file

Важно:
    Будьте осторожны с chmod -R на корневых директориях
    Не давайте права 777 на важные файлы
    Для SSH ключей используйте строгие права (600/700)
    Sticky bit на /tmp важен для безопасности
    setuid на оболочках (bash, sh) — очень опасно
    Используйте umask для контроля прав новых файлов
