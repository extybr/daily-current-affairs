#!/bin/bash
# Срез строки / массива

########## Срез строки ##########

text='Hello World'
echo "${text:1:-1}"
echo "${text::5}"

set -- 01234567890abcdefgh
echo ${1:7}

########## Срез массива ##########

mylist=(0 1 2 3 4 5)
mylist=("${mylist[@]:1}")   # Срез, начинающийся с первого элемента, кавычки обязательны
echo ${mylist[@]}           # Вывод: 1 2 3 4 5

########## Удаление элемента массива ##########

unset -v "mylist[2]"    # Удаление третьего элемента массива
echo ${mylist[@]}       # Вывод: 0 1 3 4 5

########## Удаление всего массива ##########

unset -v mylist         # Удаление списка
echo ${mylist[@]}       # Вывод:
